import AnnotationEditor, { AnnotationEditorMode } from 'components/annotation-editor';
import TestImage from './test.png';
import { useRef, useEffect, useState } from 'react';
import Button from 'components/button';

export default function  Editor() {
  const ref = useRef();
  let instance = null;
  const tools = [
    { label: '移动画布', handler: () => { instance.switchMode(AnnotationEditorMode.DragImage) } },
    { label: 'Alt 缩放画布', handler: () => { instance.switchMode(AnnotationEditorMode.Scale) } },
    { label: '矩形', handler: () => { instance.switchMode(AnnotationEditorMode.Rect) } },
    { label: '多边形', handler: () => { instance.switchMode(AnnotationEditorMode.Polyline) } },
    { label: '移动图形', handler: () => { instance.switchMode(AnnotationEditorMode.Edit) } },
    { label: '选中图形', handler: () => { instance.switchMode(AnnotationEditorMode.Select) } },
    { label: '裁剪选中图形', handler: () => { instance.clipImage() } }
  ];
  useEffect(() => {
    instance = new AnnotationEditor(ref.current);
    instance.init({
      imgUrl: TestImage,
      mode: AnnotationEditorMode.DragImage,
      select: (grp, data) => {
        console.log(data);
      }
    });
  }, []);
  return (
    <div>
      <div>
        {
          tools.map((tool, index) => {
            return (
              <Button key={index} onClick={tool.handler} style={{marginRight: '10px'}}>{tool.label}</Button>
            )
          })
        }
      </div>
      <div style={{width: '100%', height: '800px'}} ref={ref}></div>
    </div>
  )
}
